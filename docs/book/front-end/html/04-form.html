from表单

<!DOCTYPE html>
<html>
  <head>
     <title>HTML当中的表单</title>
     <meta charset="utf-8" />
  </head>
  <body>
     <h1>哈哈，我要你的数据</h1>
     <!--
       *form表单标签(必须背过)：
          属性名：       属性值：
          *action        表单提交的地址(相对/绝对)
          *method        表单提交的方法(GET/POST)
                      网页之间数据传递的方式：
                         GET(显式传值)
                           1.传递的数据会在URL地址呈现
                           2.传递长度受限
                           3.相对不安全
                           
                         POST(隐式传值)
                           1.传递的数据不会显示
                           2.传递长度不受限
                           3.相对安全
                           
       *input表单项
          属性名：       属性值：
          *type          text   文本框(默认)
                      password  密码框
                      radio     单选框(注意：name属性需要一致；value值必写)
                      checkbox  多选框(注意：name属性后需要加[]；value值必写)
                      
          *name          名称 表单该项的名称(我们需要通过名称获取相应值)
          *value         值   表单该项的值(给定就是默认值)
          *checked       默认被选中的（常用于单选和多选）
          *submit        提交按钮
          *reset         重置按钮
          
          size        文本框尺寸
          placeholder      文本框提示信息
          disabled       文本框禁用
          readonly       文本框只读
          maxlength      文本框最大输入长度设置
          
       select下拉菜单(注意：需要配合option使用)
          属性名：       属性值：
          *name          下拉菜单的名称
          
       option下拉项
          属性名：       属性值
          *value         下拉项的值
          *selected      默认被挑选出来的
          
       textarea 文本域
          属性名：       属性值：
          *name          文本域的名称
          *rows          文本域的行数(宽)
          *cols          文本域的列数(高)
          
          *注意：textarea的默认值是需要写到两个textarea标签中间的，而不是value属性中
          
     -->
     <form action="feige.php" method="get">
       账号：<input type="text" name="number" value=""/><br/><br/>
       密码：<input type="password" name="pass" value="" /><br/><br/>
       性别：<input type="radio" name="sex" value="men"/> 男
            <input type="radio" name="sex" value="women" /> 女
            <input type="radio" name="sex" value="yao" checked /> 妖<br/><br/>
       爱好：<input type="checkbox" name="hobby[]" value="game" /> 游戏
            <input type="checkbox" name="hobby[]" value="sports" checked /> 运动
            <input type="checkbox" name="hobby[]" value="sleep" /> 睡觉
            <input type="checkbox" name="hobby[]" value="eat" checked /> 吃饭
            <input type="checkbox" name="hobby[]" value="travel" /> 旅行
            <input type="checkbox" name="hobby[]" value="xijiao" checked /> 洗脚<br/><br/>
       地址：<select name="address">
              <option value="0">请选择</option>
              <option value="bj" selected >北京</option>
              <option value="tj">天津</option>
              <option value="sh">上海</option>
              <option value="cq">重庆</option>
            </select><br/><br/>
       简介：<textarea name="jianjie" rows="5" cols="30"></textarea>  <br/><br/> 
            
            <input type="submit" value="注册"/>
            <input type="reset" value="刷新"/>
     </form>
  </body>
</html>

表单当中的隐藏域 / 文件域

<!DOCTYPE html>
<html>
  <head>
     <title>HTML表单当中的隐藏域 / 文件域</title>
     <meta charset="utf-8" />
  </head>
  <body>
     <h1>*HTML表单当中的隐藏域</h1>
     <form action="feige.php" method="post" >
       
       <!--通过隐藏域传递信息(某些不希望用户输入的数据，可以使用隐藏域进行传递)-->
       <input type="hidden" name="id" value="1"/>
       
       <input type="submit" value="提交"/>
     </form>
     
    <h1>*HTML表单当中的文件域</h1>
    <!--
       文件域注意事项：
         1.必须使用post方式提交
         2.必须将input的type值设置为 file
         3.必须在form表单头部添加该属性 enctype="multipart/form-data"(文件传输通行证)
          
       8位 = 1字节 
       1024字节 = 1kb
       1024kb = 1Mb
       1024Mb = 1Gb
       1024Gb = 1Tb
     -->
     <form action="feige.php" method="post" enctype="multipart/form-data">
       选择文件：<input type="file" name="pic" /><br/><br/>
       
       <!-- <input type='submit' value="提交" /> -->
       
       <!-- 
          图片提交框：<input type="image" src="./zhuce.png" /> 
       -->
       
       <!--
          button按钮：js常用
            属性名：       属性值：
            type        submit(默认提交)
                         reset(重置)
                         button(普通按钮)
       -->
       <button type="button">提交</button>
     </form>
     <script>
       //获取按钮的元素
       var but = document.getElementsByTagName('button')[0]
       
       //给按钮元素绑定了一个鼠标移入事件
       but.onmouseover = function(){
          alert('我是一个普通的按钮，js中我比较常用！');
       }
     </script>
  </body>
</html>
